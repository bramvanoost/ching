# CHING iOS Phase 2 v2 — Minimum Playable Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Get an ugly-but-tappable CHING iOS app running on the simulator — a single screen where the human plays a full 1v1 game vs the Jones AI from start to game-over, driven by the existing `CHINGEngine` Swift Package.

**Architecture:** A new Xcode iOS app target at `ios/CHING/` consumes `ios/CHINGEngine/` as a local Swift Package dependency. A `@MainActor @Observable` `GameStore` wraps engine `State` and owns the `Mulberry32` RNG; SwiftUI views read the store and dispatch `Action`s. AI turns are driven synchronously after each human action by walking `decide` until control returns to the human or the game ends. No visual system, no animations, no other screens. Default SwiftUI styling. Verification is a real-device-style playthrough on the iPhone simulator plus a screenshot.

**Tech Stack:** Swift 5.10 / SwiftUI / iOS 17+ / Xcode 16+, `@Observable` macro for state, XCTest for unit tests, local Swift Package for engine, `xcodebuild` + `xcrun simctl` for command-line builds and screenshots.

## Engine API reference (do not re-derive)

These are the only engine symbols the iOS app calls. All live in `import CHINGEngine`:

- `func initialState(playerIds: [String]) -> State`
- `func step<R: CHINGRandom>(state: State, action: Action, rng: inout R) -> State`
- `func score(_ state: State) -> [Int]`
- `func decide(state: State, ai: Difficulty) -> Action`
- `struct Difficulty(discipline: Double)`
- `struct Mulberry32(seed: UInt32) // conforms to CHINGRandom`
- `enum Action { case roll, case pick(face: Face), case stop }`
- `enum Face: Int { case one, two, three, four, five, coin; var value: Int }`
- `enum Phase: String { case roll, pick, over }`
- `struct State { players, current, centerTiles, diceInHand, rolled, setAside, pickedFaces, phase }`
- `struct Player { id, tiles }`
- `func tileCoins(_ tile: Int) -> Int // 1..4`
- `let TOTAL_DICE = 8`

The `Action.stop` case is labelled **"Bank"** in the user-facing UI (matches the TS CLI vocabulary). The `Face.coin` case is shown as **"COIN"** or **"C"**.

The AI persona for v2 is **Jones** at `discipline: 0.30` (per `docs/superpowers/specs/2026-06-05-ching-ios-design.md`).

## Task 2: Implement `GameStore`

**Files:**
- Create: `ios/CHING/CHING/GameStore.swift`

`GameStore` is a `@MainActor @Observable` reference type that owns the engine `State` and the `Mulberry32` RNG, exposes derived computed properties for the view, dispatches `Action`s via `step`, and auto-runs AI turns until control returns to the human or the game ends.

- [ ] **Step 1: Create `GameStore.swift` with the full implementation**

Write `ios/CHING/CHING/GameStore.swift`:

```swift
import Foundation
import Observation
import CHINGEngine

@MainActor
@Observable
final class GameStore {
    static let humanSeat = 0
    static let aiSeat = 1

    private(set) var state: State
    private var rng: Mulberry32

    // Jones, the greedy persona. Discipline 0.30 per the design spec.
    private let aiDifficulty = Difficulty(discipline: 0.30)

    init(seed: UInt32) {
        self.rng = Mulberry32(seed: seed)
        self.state = initialState(playerIds: ["YOU", "JONES"])
    }

    convenience init() {
        self.init(seed: UInt32.random(in: 1...UInt32.max))
    }

    var scores: [Int] { score(state) }
    var setAsideSum: Int { state.setAside.reduce(0) { $0 + $1.value } }
    var isHumanTurn: Bool { state.current == Self.humanSeat }
    var isOver: Bool { state.phase == .over }

    var canRoll: Bool {
        state.phase == .roll && state.diceInHand > 0 && isHumanTurn
    }

    var canBank: Bool {
        state.phase == .roll && !state.setAside.isEmpty && isHumanTurn
    }

    func canPick(_ face: Face) -> Bool {
        state.phase == .pick &&
            isHumanTurn &&
            !state.pickedFaces.contains(face) &&
            state.rolled.contains(face)
    }

    func apply(_ action: Action) {
        state = step(state: state, action: action, rng: &rng)
    }

    /// Drives the AI synchronously while it's the AI's turn. Safe to call
    /// any time; no-ops when it's the human's turn or the game is over.
    func runAIIfNeeded() {
        while !isOver && !isHumanTurn {
            let action = decide(state: state, ai: aiDifficulty)
            apply(action)
        }
    }

    func newGame() {
        rng = Mulberry32(seed: UInt32.random(in: 1...UInt32.max))
        state = initialState(playerIds: ["YOU", "JONES"])
    }
}
```

```swift
import XCTest
import CHINGEngine
@testable import CHING

@MainActor
final class GameStoreTests: XCTestCase {
    func test_init_setsUpTwoPlayersHumanTurnRollPhase() {
        let store = GameStore(seed: 1)
        XCTAssertEqual(store.state.players.count, 2)
        XCTAssertEqual(store.state.players[0].id, "YOU")
        XCTAssertEqual(store.state.players[1].id, "JONES")
        XCTAssertEqual(store.state.phase, .roll)
        XCTAssertEqual(store.state.current, 0)
        XCTAssertTrue(store.isHumanTurn)
        XCTAssertFalse(store.isOver)
        XCTAssertEqual(store.scores, [0, 0])
    }

    func test_apply_rollAdvancesPhaseOrTurn() {
        let store = GameStore(seed: 1)
        store.apply(.roll)
        // With 8 dice the chance of immediate bust is near zero, but cover
        // both branches: either we're picking, or the bust path ended our turn.
        let advanced = store.state.phase == .pick || store.state.current != 0
        XCTAssertTrue(advanced)
    }

    func test_newGame_resetsState() {
        let store = GameStore(seed: 1)
        store.apply(.roll)
        store.newGame()
        XCTAssertEqual(store.state.centerTiles, Array(21...36))
        XCTAssertEqual(store.state.phase, .roll)
        XCTAssertEqual(store.state.players[0].tiles, [])
        XCTAssertEqual(store.state.players[1].tiles, [])
        XCTAssertEqual(store.state.current, 0)
    }

    func test_runAIIfNeeded_isNoOpOnHumanTurn() {
        let store = GameStore(seed: 1)
        let before = store.state
        store.runAIIfNeeded()
        XCTAssertEqual(store.state, before)
    }

    func test_fullGameTerminates() {
        // Play a synthetic game where both seats use the AI driver. Proves
        // the loop terminates and the engine reaches .over from initial state.
        let store = GameStore(seed: 1)
        var safetyLimit = 5000
        while !store.isOver && safetyLimit > 0 {
            let action = decide(state: store.state, ai: Difficulty(discipline: 0.5))
            store.apply(action)
            safetyLimit -= 1
        }
        XCTAssertTrue(store.isOver, "Game should terminate within 5000 actions")
        XCTAssertGreaterThan(safetyLimit, 0)
    }
}
```
